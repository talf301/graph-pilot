# `gp serve` Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a `gp serve` command that runs a background HTTP server with an interactive, live-updating Cytoscape.js graph dashboard and tmux-based Claude Code session management.

**Architecture:** Express server watches the vault with `fs.watch`, serves a single-page dashboard with Cytoscape.js, pushes graph state over WebSocket on changes, and exposes a REST API for triggering launch/design/dispatch actions via tmux. No build step — vanilla JS client modules served statically.

**Tech Stack:** Express, ws (WebSocket), Cytoscape.js (CDN: core + cose-bilkent + dagre extensions), Node fs.watch, tmux.

**Spec:** `docs/superpowers/specs/2026-03-26-gp-serve-design.md`

- [ ] **Step 1: Implement tmux.ts**

```typescript
import { execSync, execFileSync } from "node:child_process";

const SESSION_NAME = "graphpilot";

/** Check if tmux is installed */
export function tmuxAvailable(): boolean {
  try {
    execFileSync("tmux", ["-V"], { stdio: "pipe" });
    return true;
  } catch {
    return false;
  }
}

/** Ensure the graphpilot tmux session exists */
export function ensureSession(): void {
  try {
    execFileSync("tmux", ["has-session", "-t", SESSION_NAME], { stdio: "pipe" });
  } catch {
    // Session doesn't exist — create it detached
    execFileSync("tmux", ["new-session", "-d", "-s", SESSION_NAME], { stdio: "pipe" });
  }
}

/** Spawn a command in a new tmux window. Returns the window name. */
export function spawnWindow(name: string, command: string): string {
  // Sanitize window name (tmux doesn't like special chars)
  const windowName = name.replace(/[^a-zA-Z0-9_-]/g, "-");
  execFileSync("tmux", [
    "new-window",
    "-t", SESSION_NAME,
    "-n", windowName,
    command,
  ], { stdio: "pipe" });
  return windowName;
}

/** List active tmux windows in the session. Returns window names. */
export function listWindows(): string[] {
  try {
    const output = execFileSync("tmux", [
      "list-windows",
      "-t", SESSION_NAME,
      "-F", "#{window_name}",
    ], { stdio: "pipe", encoding: "utf-8" });
    return output.trim().split("\n").filter(Boolean);
  } catch {
    return [];
  }
}
```

- [ ] **Step 1: Implement the graph-building helper**

At the top of `serve.ts`, add a function that loads vault nodes and converts them into the JSON shape the frontend expects:

```typescript
import fs from "node:fs";
import path from "node:path";
import http from "node:http";
import { spawn } from "node:child_process";
import express from "express";
import { WebSocketServer, WebSocket } from "ws";
import { loadAllNodes, indexById, resolveRef } from "./vault.js";
import { type GpConfig, type GraphNode } from "./schema.js";
import { ensureSession, spawnWindow, tmuxAvailable } from "./tmux.js";

interface ApiNode {
  id: string;
  type: string;
  status: string;
  project: string;
  parent: string | null;
  dependsOn: string[];
  blocks: string[];
  body: string;
  filepath: string;
}

interface ApiEdge {
  source: string;
  target: string;
  type: "parent" | "depends-on";
}

interface GraphData {
  nodes: ApiNode[];
  edges: ApiEdge[];
}

function refToId(ref: string): string {
  return ref.replace(/^\[\[/, "").replace(/\]\]$/, "").toLowerCase().replace(/\s+/g, "-");
}

async function buildGraphData(vaultRoot: string): Promise<GraphData> {
  const nodes = await loadAllNodes(vaultRoot);
  const nodeIds = new Set(nodes.map(n => n.meta.id));

  const apiNodes: ApiNode[] = nodes.map(n => ({
    id: n.meta.id,
    type: n.meta.type,
    status: n.meta.status,
    project: n.meta.project,
    parent: n.meta.parent ? refToId(n.meta.parent) : null,
    dependsOn: (n.meta["depends-on"] ?? []).map(refToId),
    blocks: (n.meta.blocks ?? []).map(refToId),
    body: n.body,
    filepath: n.filepath,
  }));

  const apiEdges: ApiEdge[] = [];
  for (const n of nodes) {
    // Parent edges: arrow points child → parent (child is source, parent is target)
    if (n.meta.parent) {
      const parentId = refToId(n.meta.parent);
      if (nodeIds.has(parentId)) {
        apiEdges.push({ source: n.meta.id, target: parentId, type: "parent" });
      }
    }
    // Depends-on edges: arrow points from node → its dependency
    for (const dep of n.meta["depends-on"] ?? []) {
      const depId = refToId(dep);
      if (nodeIds.has(depId)) {
        apiEdges.push({ source: n.meta.id, target: depId, type: "depends-on" });
      }
    }
  }

  return { nodes: apiNodes, edges: apiEdges };
}
```

- [ ] **Step 2: Implement the Express app and WebSocket server**

```typescript
export interface ServeOptions {
  vaultRoot: string;
  config: GpConfig;
  port: number;
  host: string;
}

export async function startServer(opts: ServeOptions): Promise<void> {
  const { vaultRoot, config, port, host } = opts;

  if (!tmuxAvailable()) {
    console.error("Error: tmux is not installed. gp serve requires tmux.");
    process.exit(1);
  }

  ensureSession();

  const app = express();
  app.use(express.json());

  // Serve static files from public/
  const publicDir = path.join(path.dirname(new URL(import.meta.url).pathname), "..", "public");
  app.use(express.static(publicDir));

  // --- REST API ---

  app.get("/api/graph", async (_req, res) => {
    try {
      const data = await buildGraphData(vaultRoot);
      res.json(data);
    } catch (err: unknown) {
      res.status(500).json({ error: err instanceof Error ? err.message : String(err) });
    }
  });

  app.get("/api/node/:id", async (req, res) => {
    try {
      const nodes = await loadAllNodes(vaultRoot);
      const index = indexById(nodes);
      const node = resolveRef(req.params.id, index);
      if (!node) {
        res.status(404).json({ error: `Node not found: ${req.params.id}` });
        return;
      }
      res.json({
        id: node.meta.id,
        type: node.meta.type,
        status: node.meta.status,
        project: node.meta.project,
        parent: node.meta.parent ? refToId(node.meta.parent) : null,
        dependsOn: (node.meta["depends-on"] ?? []).map(refToId),
        blocks: (node.meta.blocks ?? []).map(refToId),
        body: node.body,
        filepath: node.filepath,
        artifacts: node.meta.artifacts,
      });
    } catch (err: unknown) {
      res.status(500).json({ error: err instanceof Error ? err.message : String(err) });
    }
  });

  app.post("/api/launch/:id", async (req, res) => {
    try {
      const windowName = spawnWindow(req.params.id, `gp launch ${req.params.id}`);
      res.json({ ok: true, window: windowName });
    } catch (err: unknown) {
      res.status(500).json({ error: err instanceof Error ? err.message : String(err) });
    }
  });

  app.post("/api/design", async (_req, res) => {
    try {
      const windowName = spawnWindow("design", "gp design");
      res.json({ ok: true, window: windowName });
    } catch (err: unknown) {
      res.status(500).json({ error: err instanceof Error ? err.message : String(err) });
    }
  });

  app.post("/api/dispatch/:id", async (req, res) => {
    try {
      const nodes = await loadAllNodes(vaultRoot);
      const index = indexById(nodes);
      const node = resolveRef(req.params.id, index);
      if (!node) {
        res.status(404).json({ error: `Node not found: ${req.params.id}` });
        return;
      }

      // Try frontmatter first, then request body
      const planId = node.meta.artifacts?.["dispatch-run"] ?? req.body?.planId;
      if (!planId) {
        res.status(400).json({ error: "No planId found in node frontmatter or request body" });
        return;
      }

      const windowName = spawnWindow(
        `${req.params.id}-dispatch`,
        `gp dispatch ${req.params.id} --plan ${planId}`
      );
      res.json({ ok: true, window: windowName });
    } catch (err: unknown) {
      res.status(500).json({ error: err instanceof Error ? err.message : String(err) });
    }
  });

  // --- HTTP + WebSocket ---

  const server = http.createServer(app);
  const wss = new WebSocketServer({ server });

  // Track connected clients
  const clients = new Set<WebSocket>();
  wss.on("connection", async (ws) => {
    clients.add(ws);
    ws.on("close", () => clients.delete(ws));

    // Send initial graph state
    try {
      const data = await buildGraphData(vaultRoot);
      ws.send(JSON.stringify({ type: "graph-update", ...data }));
    } catch {
      // Best effort
    }
  });

  // --- File watching ---

  const projectsDir = path.join(vaultRoot, config.root);
  let debounceTimer: ReturnType<typeof setTimeout> | null = null;

  async function pushGraphUpdate() {
    const start = Date.now();
    try {
      const data = await buildGraphData(vaultRoot);
      const msg = JSON.stringify({ type: "graph-update", ...data });
      for (const ws of clients) {
        if (ws.readyState === WebSocket.OPEN) {
          ws.send(msg);
        }
      }
      const elapsed = Date.now() - start;
      if (elapsed > 200) {
        console.warn(`[gp serve] graph rebuild + push took ${elapsed}ms (${data.nodes.length} nodes) — consider diffing`);
      }
    } catch (err) {
      console.error("[gp serve] error rebuilding graph:", err);
    }
  }

  fs.watch(projectsDir, { recursive: true }, (eventType, filename) => {
    if (!filename?.endsWith(".md")) return;
    if (debounceTimer) clearTimeout(debounceTimer);
    debounceTimer = setTimeout(pushGraphUpdate, 300);
  });

  // --- Start ---

  server.listen(port, host, () => {
    const url = `http://${host === "0.0.0.0" ? "localhost" : host}:${port}`;
    console.log(`GraphPilot dashboard: ${url}`);

    // Signal parent process that we're ready (for daemon mode)
    if (process.send) {
      process.send({ type: "ready", url });
    }
  });
}
```

- [ ] **Step 3: Add daemon helpers at the bottom of serve.ts**

```typescript
const GP_DIR = path.join(process.env.HOME ?? "~", ".graphpilot");
const PID_FILE = path.join(GP_DIR, "serve.pid");
const LOG_FILE = path.join(GP_DIR, "serve.log");

export function daemonize(opts: ServeOptions): void {
  fs.mkdirSync(GP_DIR, { recursive: true });

  // Check if already running
  if (fs.existsSync(PID_FILE)) {
    const pid = parseInt(fs.readFileSync(PID_FILE, "utf-8").trim(), 10);
    try {
      process.kill(pid, 0); // Check if process exists
      console.log(`gp serve is already running (PID ${pid}). Use 'gp serve --stop' to stop it.`);
      process.exit(1);
    } catch {
      // Process not running, clean up stale PID file
      fs.unlinkSync(PID_FILE);
    }
  }

  const logFd = fs.openSync(LOG_FILE, "a");
  const child = spawn(process.execPath, [
    path.join(path.dirname(new URL(import.meta.url).pathname), "serve-worker.js"),
    JSON.stringify(opts),
  ], {
    detached: true,
    stdio: ["ignore", logFd, logFd, "ipc"],
    env: process.env,
  });

  child.on("message", (msg: any) => {
    if (msg?.type === "ready") {
      fs.writeFileSync(PID_FILE, String(child.pid), "utf-8");
      console.log(`GraphPilot dashboard: ${msg.url}`);
      console.log(`PID ${child.pid} — logs at ${LOG_FILE}`);
      child.unref();
      child.disconnect();
      fs.closeSync(logFd);
      process.exit(0);
    }
  });

  // Timeout if server doesn't start
  setTimeout(() => {
    console.error("Server failed to start within 10 seconds. Check logs at", LOG_FILE);
    child.kill();
    fs.closeSync(logFd);
    process.exit(1);
  }, 10000);
}

export function stopDaemon(): void {
  if (!fs.existsSync(PID_FILE)) {
    console.log("gp serve is not running.");
    return;
  }

  const pid = parseInt(fs.readFileSync(PID_FILE, "utf-8").trim(), 10);
  try {
    process.kill(pid, "SIGTERM");
    fs.unlinkSync(PID_FILE);
    console.log(`Stopped gp serve (PID ${pid}).`);
  } catch {
    fs.unlinkSync(PID_FILE);
    console.log("Process was not running. Cleaned up PID file.");
  }
}
```

Note: `daemonize` spawns a separate entry point `serve-worker.js` which is just a tiny wrapper that calls `startServer`. This avoids the daemon process inheriting the CLI's process.argv logic.

```typescript
async function cmdServe(args: string[]) {
  const { config, vaultRoot } = requireConfig();
  const { flags } = parseFlags(args);

  if (args.includes("--stop")) {
    stopDaemon();
    return;
  }

  const port = parseInt(flags.port?.[0] ?? "4800", 10);
  const host = flags.host?.[0] ?? "localhost";

  const opts = { vaultRoot, config, port, host };

  if (args.includes("--foreground")) {
    // Run in foreground (useful for development)
    await startServer(opts);
  } else {
    daemonize(opts);
  }
}
```

```javascript
    const node = selectedNodeData;

    html += `<div class="detail-type type-${node.type}">${node.type.toUpperCase()}</div>`;
    html += `<div class="detail-id">${node.id}</div>`;
    html += `<div class="detail-status status-${node.status}">● ${node.status}</div>`;

    // Body (first 300 chars as preview)
    if (node.body && node.body.trim()) {
      const preview = node.body.trim().slice(0, 500);
      html += `<div class="detail-body">${escapeHtml(preview)}</div>`;
    }

    // Parent
    if (node.parent) {
      html += `<div class="detail-section">`;
      html += `<div class="detail-section-label">PARENT</div>`;
      html += `<div class="detail-link" onclick="selectNode('${node.parent}')">↑ ${node.parent}</div>`;
      html += `</div>`;
    }

    // Dependencies
    if (node.dependsOn && node.dependsOn.length > 0) {
      html += `<div class="detail-section">`;
      html += `<div class="detail-section-label">DEPENDS ON</div>`;
      for (const dep of node.dependsOn) {
        const depNode = cy.getElementById(dep);
        const depStatus = depNode.length ? depNode.data("status") : "unknown";
        const icon = depStatus === "done" ? "✓" : "○";

      html += `</div>`;

    // Children (nodes whose parent is this node)
    // Edge direction: child → parent (source=child, target=parent)
    const children = cy.nodes().filter((n) => {
```
